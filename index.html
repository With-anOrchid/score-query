    <script>

        // 加载成绩数据

        let scoresData = null;

        

        fetch('scores.json')

            .then(response => response.json())

            .then(data => {

                scoresData = data;

            })

            .catch(error => {

                console.error('加载数据失败:', error);

            });


        // 处理表单提交

        document.getElementById('searchForm').addEventListener('submit', function(e) {

            e.preventDefault();

            

            const name = document.getElementById('name').value.trim();

            const id = document.getElementById('id').value.trim();

            

            if (!scoresData) {

                alert('数据正在加载中，请稍后再试');

                return;

            }

            

            // 查找学生

            const student = scoresData.students.find(s => 

                s.name === name && s.id === id

            );

            

            const resultSection = document.getElementById('resultSection');

            const studentInfo = document.getElementById('studentInfo');

            const scoreBody = document.getElementById('scoreBody');

            

            if (student) {

                // 显示学生信息

                studentInfo.innerHTML = `

                    <h3>${student.name} 同学</h3>

                    <p>学号: ${student.id}</p>

                    <p style="font-size: 1.2em; margin-top: 10px;">总分: <strong>${student.total}</strong></p>

                `;

                

                // 显示成绩详情

                scoreBody.innerHTML = '';

                const questions = ['q3', 'q4', 'q5', 'q6', 'q7', 'q8', 'q9', 'q10'];

                

                questions.forEach(q => {

                    const questionNum = q.substring(1);

                    const studentAnswer = student.scores[q].option;

                    const studentScore = student.scores[q].score;

                    const correctAnswer = scoresData.correctAnswers[q];

                    const isCorrect = studentAnswer === correctAnswer;

                    

                    const row = document.createElement('tr');

                    row.innerHTML = `

                        <td>第${questionNum}题</td>

                        <td class="${isCorrect ? 'correct' : 'incorrect'}">${studentAnswer}</td>

                        <td>${studentScore}</td>

                        <td class="correct">${correctAnswer}</td>

                        <td class="${isCorrect ? 'correct' : 'incorrect'}">

                            ${isCorrect ? '正确' : '错误'}

                        </td>

                    `;

                    scoreBody.appendChild(row);

                });

                

                // 添加总分行

                const totalRow = document.createElement('tr');

                totalRow.className = 'total-score';

                totalRow.innerHTML = `

                    <td colspan="2">总分</td>

                    <td colspan="3">${student.total}</td>

                `;

                scoreBody.appendChild(totalRow);

                

                resultSection.style.display = 'block';

                

                // 滚动到结果区域

                resultSection.scrollIntoView({ behavior: 'smooth' });

                

            } else {

                resultSection.style.display = 'block';

                studentInfo.innerHTML = '';

                scoreBody.innerHTML = `

                    <tr>

                        <td colspan="5" class="error-message">

                            未找到匹配的学生信息，请检查姓名和学号是否正确

                        </td>

                    </tr>

                `;

            }

        });
